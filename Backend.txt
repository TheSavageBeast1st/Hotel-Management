create table cost (S_No int auto_increment primary key, Room_No int, Suite varchar(30) Not NULL, No_Of_Days int , Total_Cost decimal(15,2), foreign key (Room_No) references Baratie_Booking(Room_No) on update cascade, foreign key (Suite) references suite_pricing(Suite) on update cascade);

insert into cost (Room_No, Suite, No_Of_Days, Total_Cost)
select
    b.Room_No,
    b.Suite,
    b.No_Of_Days,
    sp.Cost_Per_Night * b.No_Of_Days AS Total_Cost
from 
    Baratie_Booking b
join 
    suite_pricing sp ON b.Suite = sp.Suite;
